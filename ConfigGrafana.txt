SELECT
  timestamp AS time,
  cpu_percent AS "CPU %",
  n.name AS node
FROM metrics m
JOIN nodes n ON n.id = m.node_id
WHERE timestamp >= NOW() - INTERVAL '5 minutes'
ORDER BY timestamp ASC
